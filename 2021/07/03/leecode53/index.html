使用动态规划

添加了输出子序列

/***使用动态规划，求数组中nums最大子序列的和；最终输出子序列***/
#include<iostream>
using namespace std;
int main()
{
    int nums[] = { -2, 1, -3, 4, -1, 2, 1, -5, 4 };
    
    const int n = sizeof(nums)/sizeof(n);
    int nums_copy[n] = {};
    memcpy(nums_copy, nums,sizeof(nums));
    int max_data = nums[0];
    for (int i = 1; i < n; i++)
    {
        if (nums[i - 1] > 0)
        {
            nums[i] += nums[i - 1];
        }
    }
    //找到最大的和，与其值；
    int k = 0;
    for (int i = 0; i < n; i++)
    {
        
        if (max_data < nums[i])
        {
            max_data = nums[i];
            k = i;
        }
    }
    cout << endl;
    int t = 0;
    for (int i = k;; i--)
    {
        if (nums[i] < 0)
        {
            t = i;
            break;
        }
    }
    cout << max_data << endl;
    //输出最大和的子序列
    for (int i = t + 1; i <= k; i++)
        cout << nums_copy[i] << " ";
    cout << endl;
    return 0;
}
